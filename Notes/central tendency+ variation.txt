#measures of variations or different measures of dispersions,
#they help in measuring the spread and scatterdness of the 
#data around any point, and preferably the arithmetic mean 
#value. Now, there are different types of measures which are 
#available in statistics. range,
#interquartile range, quartile deviation, absolute mean 
#deviation, variance, standard deviation

#Range is defined as the difference between the maximum and minimum value
max(time)-min(time)
max(time_NA,na.rm=T)-min(time_NA,na.rm=T)
#in-built 'range()' gives both maximum and minimum value
range(time)
range(time_NA)
range(time_NA,na.rm=T)

#interquartile range is the difference between the third quartile point and first quartile point
IQR(time)
IQR(time_NA,na.rm=T)
#quartile deviation
#half of IQR
IQR(time)/2
IQR(time_NA,na.rm=T)/2

# absolute deviation about a point
mean(abs(time-50))
mean(abs(time_NA-50),na.rm=T)

#absolute mean deviaton 
mean(abs(time-median(time)))
mean(abs(time_NA-median(time_NA,na.rm=T)),na.rm=T)
#mean absolute deviation about mean
mean(abs(time-mean(time)))


#variance 
var(time)
var(time_NA,na.rm=T)
#Standard deviation
sqrt(var(time))
#coefficient of variation
sqrt(var(time))/mean(time)
#sometimes coefficient of variation is given in % then 
sqrt(var(time))/mean(time)*100
