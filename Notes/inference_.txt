unbiasednormal=function(n,rep){
  out=matrix(nrow=rep,ncol=4,data=0)
  #In R, setting data=0 in the matrix() function creates a
  #matrix where all elements are initialized with the value 0.
  #This is a common method for creating a "zero matrix" 
  #of a specified size.
  colnames(out)=c("Mean","Variance","Median","Sum_dev_frm_M")
  mu=10
  sigma2=25
  for(r in 1:rep){
    x=rnorm(n,mu,sqrt(sigma2))
    out[r,1]=mean(x)
    out[r,2]=var(x)
    out[r,3]=median(x)
    out[r,4]=sum(x-median(x)^2)
  }
  out
  cat("Expectation of sample mean is ",mean(out[,1]),"\n")
  cat("Expectation of sample variance is ",mean(out[,2]),"\n")
  cat("Expectation of sample median is ",mean(out[,3]),"\n")
  cat("Expectation of sum(x-median(x)^2) is ",mean(out[,4]),"\n")
}


unbiasednormal(100,3)

varnormal=function(n,rep){
  out=matrix(nrow=rep,ncol=3,data=0)
  mu=10
  sigma2=25
  for(r in 1:rep){
    x=rnorm(n,mu,sqrt(sigma2))
    out[r,1]=mean(x)
    out[r,2]=x[1]
    out[r,3]=(x[1]+x[2])/2
  }
  cat(var(out[,1]),var(out[,2]),"\n")
}
